CREATE TABLE Vote (id integer NOT NULL PRIMARY KEY AUTOINCREMENT, politicianId integer, votersId integer, FOREIGN KEY (politicianId) REFERENCES Politician(id), FOREIGN KEY (votersId) REFERENCES voters(id));

CREATE TABLE Voters (id integer NOT NULL PRIMARY KEY AUTOINCREMENT, first_name varchar, last_name varchar, gender varchar, age integer);
