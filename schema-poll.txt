CREATE TABLE Candidate (
	id INTEGER PRIMARY KEY,
	name TEXT,
	partai TEXT,
	location TEXT,
	grade_current INTEGER
);

CREATE TABLE Voters (
	id INTEGER PRIMARY KEY,
	first_name TEXT,
	last_name TEXT,
	gender TEXT
);

CREATE TABLE Vote (
	id INTEGER PRIMARY KEY,
	voting_Date DATE,
	voters_id INTEGER,
	candidate_id INTEGER	
);
